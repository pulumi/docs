{{/* IMPORTANT: The referenced asset must exist in assets/fingerprinted/. Missing assets cause a build panic. */}} {{/* Strip a leading "/" so frontmatter paths like "/images/foo.svg" resolve correctly. */}} {{ $src := .src | strings.TrimPrefix "/" }} {{ $img := resources.Get (printf "fingerprinted/%s" $src) }} {{ if ne $img.MediaType.SubType "svg" }} {{ $maxWidth := .maxWidth | default 800 }} {{/* Default to q90; raise via `quality` for screenshots / images with fine text where lossy artifacts are visible. */}} {{ $quality := .quality | default 90 }} {{ $processArgs := printf "webp q%d" $quality }} {{/* Cap the 1x `src` at maxWidth, but generate srcset entries up to 2x maxWidth so high-DPI displays get sharp images (the rendered slot is `maxWidth` CSS pixels, so a 2x device needs a 2x-resolution resource). */}} {{ if gt $img.Width $maxWidth }} {{ $img = $img.Process (printf "resize %dx %s" $maxWidth $processArgs) }} {{ else }} {{ $img = $img.Process $processArgs }} {{ end }} {{/* Skip srcset for very small images (maxWidth < 400) — src already covers 1x and 2x. */}} {{ $srcset := slice }} {{ if ge $maxWidth 400 }} {{ $maxSrcsetWidth := mul $maxWidth 2 }} {{ $candidates := slice 400 600 800 1200 1600 2000 2400 3200 }} {{ $origImg := resources.Get (printf "fingerprinted/%s" $src) }} {{ range $candidates }} {{ if and (le . $maxSrcsetWidth) (ge $origImg.Width .) }} {{ $resized := $origImg.Process (printf "resize %dx %s" . $processArgs) | fingerprint }} {{ $srcset = $srcset | append (printf "%s %dw" $resized.RelPermalink .) }} {{ end }} {{ end }} {{/* If the original is at or below 2x maxWidth, include the original width too so the sharpest available resource is offered (the largest candidate may fall short of the original width — e.g. a 1472w source with a 1200 candidate). Skip when the original width exactly matches a candidate, otherwise the loop above already emitted it. */}} {{ if and (le $origImg.Width $maxSrcsetWidth) (not (in $candidates $origImg.Width)) }} {{ $resized := $origImg.Process $processArgs | fingerprint }} {{ $srcset = $srcset | append (printf "%s %dw" $resized.RelPermalink $origImg.Width) }} {{ end }} {{ end }} {{ $img = $img | fingerprint }} {{ else }} {{ $img = $img | fingerprint }} {{ end }}
