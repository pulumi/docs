Use this data source to get a list of EBS Snapshot IDs matching the specified
criteria.

{{% examples %}}
## Example Usage
{{% example %}}

```typescript
import * as pulumi from "@pulumi/pulumi";
import * as aws from "@pulumi/aws";

const ebsVolumes = pulumi.output(aws.ebs.getSnapshotIds({
    filters: [
        {
            name: "volume-size",
            values: ["40"],
        },
        {
            name: "tag:Name",
            values: ["Example"],
        },
    ],
    owners: ["self"],
}, { async: true }));
```

{{% /example %}}
{{% /examples %}}
